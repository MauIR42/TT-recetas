void writeIns8(unsigned char ins, char *is_char); -> escribe una instrucción al LCD por DB4 a DB7
void pulse_enable(); -> manda un pulso al LCD por enable para que lea 4 bits de informaci´on
void writeWord(char * word); -> escribe n caracteres en formato ascii en el LCD utilizando writeIns8
void change_display_line(int line); -> cambia entre la línea 1 y 2 del LCD de 16 x 2
void clear_display(); -> escribe la instrucción 0x1 al LCD con writeIns8
void clear_display_line(int line); -> escribe 0x20 en toda la linea seleccionada para limpiarla
void ini_LCD(); -> manda los comandos de inicialización de la LCD por medio de wirteins8, enciende la LCD y limpia la LCD.
